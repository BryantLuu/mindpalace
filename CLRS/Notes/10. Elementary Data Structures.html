Queue

Insert: enqueue
Delete: dequeue

has head and tail.

public class Queue {
    int tail;
    int head;

    public void enqueue(double x) {
        Q[tail] = x;
        // can use use tail = (tail + 1) % Q.length;
        if (tail == Q.length - 1) {
            tail = 0;
        } else {
            tail += 1;
        }
    }

    public void dequeue() {
        x = Q[head];

        if (head == Q.length - 1) {
            head = 0;
        } else {
            head += 1;
        }
        return x;
    }
}
